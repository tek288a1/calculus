\author{Jim Fowler \and Bart Snapp}

\outcome{Compute gradients.}

\begin{document}

\makerandom

\begin{sagesilent}
  variables = [var('x'), var('y'), var('z')]
  f = sum([randint(-4,4) * prod([v**randint(0,2) for v in variables]) for _ in range(3)])
  a = randint(-1,1)
  b = randint(-1,1)
  c = randint(-1,1)
  gradient = [derivative(f,v) for v in variables]
\end{sagesilent}

\begin{exercise}

  Let $F(x,y,z) = \sage{f}$.

  \begin{hint}
    Take partial derivatives to find the components of $\grad F(x,y,z)$.
  \end{hint}  
  
  Then $\grad F(\sage{a},\sage{b},\sage{c}) = \vector{\answer{\sage{gradient[0](x=a,y=b,z=c)}}, \answer{\sage{gradient[1](x=a,y=b,z=c)}}, \answer{\sage{gradient[2](x=a,y=b,z=c)}}}$.
\end{exercise}
